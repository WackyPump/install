#删除firefox
sudo snap remove firefox 

wget https://download-installer.cdn.mozilla.net/pub/firefox/releases/142.0/linux-x86_64/zh-CN/firefox-142.0.tar.xz
# 创建软连接到 /usr/local/bin	sudo ln -s /opt/firefox/firefox /usr/local/bin/firefox
# 或者直接在bashrc文件中导入路径 export PATH=/opt/firefox:$PATH
# 或者直接在bashrc文件中使用别名 alias firefox=/opt/firefox/firefox
	创建快捷方式
	sudo nano ~/.local/share/applications/firefox.desktop
	[Desktop Entry]
	Name=Firefox
	Exec=/opt/firefox/firefox
	Icon=/opt/firefox/browser/chrome/icons/default/default128.png
	Type=Application
	Categories=Network
